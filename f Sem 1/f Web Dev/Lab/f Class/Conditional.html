<!doctype HTML>

<html>

  <head></head>

  <body>

    <script>

      let num = prompt("Enter a number less than 5 or greater than 20");

      num = Number(num);


      if (num < 5) {

        document.write("Number is less than 5! <br>");

      } else if (num > 20) {

        document.write("Number is greater than 20! <br>");

      } else {

        document.write("Number is between 5 and 20 (invalid input). <br>");

      }

    </script>

  </body>

</html>
